103 0xA8, 0x3F, // Set multiplex (HEIGHT-1): 0x3F for 128x64

107 0xA8, 0x1F, // Set multiplex (HEIGHT-1): 0x1F for 128x32

114 0xD5, 0xF0, // Set display clock divide and frequency set clock to max
115 0x2E, // Deactivate scroll

157// Print a character
158void ssd1306_WriteChar(char ch, uint8_t fsize) {
159 uint8_t sliceChar, shifter;
160 uint16_t offsetChar; // calculated position of character in font array
161 uint32_t temp;
162
163 fsize = (fsize & 0x03) + 1; // Prevent array overflow
164 if ((ch < 32) || (ch > 100)) {
165 ch = 32; // Prevent to search outside of chars array
166 }
167
168 for (uint8_t k = 0; k < fsize; k++) {
169 offsetChar = (ch - 32) * 5;
170 for (sliceChar = 0; sliceChar < (5 * fsize); sliceChar += fsize) {
171 temp = 0;
172
173 for (uint8_t j = 0; j < 8; j++) { // This routine is designed to multiply the pixels in vertical based on size
174 shifter = j * (fsize - 1); // Set the shift amount to spread the pixels on multiple pages based on size
175 for (uint8_t m = 0; m < fsize; m++) {
176 temp |= ((font_5x8[offsetChar] & (0x01 << j)) << (shifter + m));
177 }
178 }
179 shifter = 8 * k; // Set the shifting amount to select the right portion of the char
180 for (uint8_t m = 0; m < fsize; m++) {
181 i2cBuff[sliceChar + m] = (uint8_t) (temp >> shifter);
182 }
183 offsetChar++;
184 }
185 for (uint8_t m = 0; m < fsize; m++) {
186 i2cBuff[sliceChar++] = 0x00;
187 }
188 i2cWrite(SSD1306_I2C_DATA, i2cBuff, sliceChar);
189 ssd1306_SetCursor(col, ++page);
190 }
191 ssd1306_SetCursor(col += (fsize * SSD1306_CHAR_WIDTH), page -= fsize);
192
193}

204// Set cursor position - Vertical value is with increments of 8 pixels
205void ssd1306_SetCursor(uint8_t xpos, uint8_t ypos) {
206 col = xpos & 0x7F; //Prevent column overflow
207#if (SSD1306_HEIGHT == 64)
208 page = ypos & 0x07; //Prevent rows overflow if 8 rows (64 pixel height)
209#else
210 page = ypos & 0x03; //Prevent rows overflow if 4 rows (32 pixel height)
211#endif
212 i2cBuff[0] = col & 0x0F; // set low nibble of start column
213 i2cBuff[1] = 0x10 | (col >> 4); // set high nibble of start column
214 i2cBuff[2] = 0xB0 | page; // set start page
215 i2cWrite(SSD1306_I2C_CMD, i2cBuff, 3);
216}
