392static void handleSetRtcBtns(uint32_t eventId, displayCtx_t *ctx) {
393 switch (eventId) {
394 case DISP_EV_BTN_SET:
395 displayCursorSetTime(ctx->digitCursor, 32);
396 ctx->digitCursor++;
397 break;
398
399 case DISP_EV_BTN_INC:
400 if ((ctx->digitCursor == TEEN_HRS) && (ctx->showTime[ctx->digitCursor] < 2)) {
401 ctx->showTime[ctx->digitCursor]++;
402 if ((ctx->showTime[ctx->digitCursor] == 2) && (ctx->showTime[ctx->digitCursor + 1] > 3)) {
403 ctx->showTime[ctx->digitCursor + 1] = 3;
404 }
405 }
406 if (((ctx->digitCursor == UNIT_HRS) && (ctx->showTime[ctx->digitCursor] < 9) && (ctx->showTime[TEEN_HRS] < 2))
407 || ((ctx->digitCursor == UNIT_HRS) && (ctx->showTime[ctx->digitCursor] < 3) && (ctx->showTime[TEEN_HRS] == 2))) {
408 ctx->showTime[ctx->digitCursor]++;
409 }
410 if ((ctx->digitCursor == TEEN_MINS) && (ctx->showTime[ctx->digitCursor] < 5)) {
411 ctx->showTime[ctx->digitCursor]++;
412 }
413 if ((ctx->digitCursor == UNIT_MINS) && (ctx->showTime[ctx->digitCursor] < 9)) {
414 ctx->showTime[ctx->digitCursor]++;
415 }
416 break;
417
418 case DISP_EV_BTN_DEC:
419 if (ctx->showTime[ctx->digitCursor] > 0) {
420 ctx->showTime[ctx->digitCursor]--;
421 }
422 break;
423
424 default:
425 break;
426 }
427
428 if (ctx->digitCursor == 4) {
429 // All digits set update RTC and start sync
430 RTC_Time.Hours = (ctx->showTime[TEEN_HRS] * 10) + ctx->showTime[UNIT_HRS];
431 RTC_Time.Minutes = (ctx->showTime[TEEN_MINS] * 10) + ctx->showTime[UNIT_MINS];
432 RTC_Time.Seconds = 0;
433 RTC_Date.Date = 1;
434 RTC_Date.Month = RTC_MONTH_JANUARY;
435 RTC_Date.Year = 21;
436 RTC_Date.WeekDay = RTC_WEEKDAY_FRIDAY;
437
438 taskENTER_CRITICAL();
439 HAL_RTC_SetTime(hrtcHandle, &RTC_Time, RTC_FORMAT_BIN);
440 HAL_RTC_SetDate(hrtcHandle, &RTC_Date, RTC_FORMAT_BIN);
441 taskEXIT_CRITICAL();
442
443 ctx->state = DISP_SYNC;
444 xTaskNotify(clockTaskHandle, (uint32_t) CLOCK_EV_NEW_TIME, eSetValueWithOverwrite);
445 } else {
446 displayShowClock(ctx->showTime);
447 displayCursorSetTime(ctx->digitCursor, 96);
448 }
449}

469static void handleSetSilentBtns(uint32_t eventId, displayCtx_t *ctx, char *buf) {
470 switch (eventId) {
471 case DISP_EV_BTN_SET:
472 displayCursorSetTime(ctx->digitCursor, 32);
473 ctx->digitCursor++;
474 break;
475
476 case DISP_EV_BTN_INC: {
477 uint8_t tensIdx = (ctx->digitCursor < 2) ? 0 : 2;
478 if (ctx->digitCursor == tensIdx) {
479 // Tens digit: 0-2
480 if (ctx->showTime[ctx->digitCursor] < 2) {
481 ctx->showTime[ctx->digitCursor]++;
482 if (ctx->showTime[ctx->digitCursor] == 2 && ctx->showTime[ctx->digitCursor + 1] > 3) {
483 ctx->showTime[ctx->digitCursor + 1] = 3;
484 }
485 }
486 } else {
487 // Units digit: 0-9 (tens<2) or 0-3 (tens==2)
488 uint8_t max = (ctx->showTime[tensIdx] == 2) ? 3 : 9;
489 if (ctx->showTime[ctx->digitCursor] < max) {
490 ctx->showTime[ctx->digitCursor]++;
491 }
492 }
493 break;
494 }
495
496 case DISP_EV_BTN_DEC:
497 if (ctx->showTime[ctx->digitCursor] > 0) {
498 ctx->showTime[ctx->digitCursor]--;
499 }
500 break;
501
502 default:
503 break;
504 }
505
506 if (ctx->digitCursor == 4) {
507 // Save to backup registers and persist to Flash
508 setSilentHours(ctx->showTime[0] * 10 + ctx->showTime[1],
509 ctx->showTime[2] * 10 + ctx->showTime[3]);
510 flashWriteSettings();
511 if (ctx->setupMode) {
512 enterSetCorrection(ctx, buf);
513 } else {
514 ssd1306_ClearScreen();
515 ctx->state = DISP_CLOCK;
516 }
517 } else {
518 displayShowSilentHours(ctx->showTime);
519 displayCursorSetTime(ctx->digitCursor, 96);
520 }

542static void handleSetCorrBtns(uint32_t eventId, displayCtx_t *ctx, char *buf) {
543 switch (eventId) {
544 case DISP_EV_BTN_SET:
545 displayCursorSetTime(ctx->digitCursor, 32);
546 ctx->digitCursor++;
547 break;
548
549 case DISP_EV_BTN_INC:
550 if (ctx->digitCursor == 0) {
551 ctx->showTime[0] = !ctx->showTime[0];
552 } else if (ctx->digitCursor == 1) {
553 // Hundreds: 0-5, clamp lower digits to keep <= 511
554 if (ctx->showTime[1] < 5) {
555 ctx->showTime[1]++;
556 if (ctx->showTime[1] == 5) {
557 if (ctx->showTime[2] > 1) ctx->showTime[2] = 1;
558 if (ctx->showTime[2] == 1 && ctx->showTime[3] > 1) ctx->showTime[3] = 1;
559 }
560 }
561 } else if (ctx->digitCursor == 2) {
562 uint8_t max = (ctx->showTime[1] == 5) ? 1 : 9;
563 if (ctx->showTime[2] < max) {
564 ctx->showTime[2]++;
565 if (ctx->showTime[1] == 5 && ctx->showTime[2] == 1 && ctx->showTime[3] > 1) {
566 ctx->showTime[3] = 1;
567 }
568 }
569 } else {
570 uint8_t max = (ctx->showTime[1] == 5 && ctx->showTime[2] == 1) ? 1 : 9;
571 if (ctx->showTime[3] < max) {
572 ctx->showTime[3]++;
573 }
574 }
575 break;
576
577 case DISP_EV_BTN_DEC:
578 if (ctx->digitCursor == 0) {
579 ctx->showTime[0] = !ctx->showTime[0];
580 } else {
581 if (ctx->showTime[ctx->digitCursor] > 0) {
582 ctx->showTime[ctx->digitCursor]--;
583 }
584 }
585 break;
586
587 default:
588 break;
589 }
590
591 if (ctx->digitCursor == 4) {
592 // Save to backup register, persist to Flash, and apply immediately
593 uint16_t value = ctx->showTime[1] * 100 + ctx->showTime[2] * 10 + ctx->showTime[3];
594 setCalibration(ctx->showTime[0], value);
595 applyCalibration();
596 flashWriteSettings();
597 if (ctx->setupMode) {
598 enterSetRtc(ctx, buf);
599 } else {
600 ssd1306_ClearScreen();
601 ctx->state = DISP_CLOCK;
602 }
603 } else {
604 displayShowCalibration(ctx->showTime);
605 displayCursorSetTime(ctx->digitCursor, 96);
606 }
607}

651void displayTask(void *parameters) {
652 char buf[11];
653 uint32_t eventId;
654 displayCtx_t ctx = {
655 .state = DISP_SYNC,
656 .digitCursor = 0,
657 .isOn = OFF,
658 .lastOnTime = xTaskGetTickCount(),
659 .lastClockUpdate = xTaskGetTickCount(),
660 };
661
662 while (1) {
663 if (xTaskNotifyWait(0, 0xffffffff, &eventId, pdMS_TO_TICKS(DISPLAY_TASK_DELAY)) == pdTRUE) {
664
665 // First boot full setup: silent hours calibration time
666 if (eventId == DISP_EV_FORCE_SETUP) {
667 ctx.setupMode = 1;
668 enterSetSilent(&ctx, buf);
669 displayOnOff(ON, &ctx);
670 vTaskResume(buttonTaskHandle);
671 continue;
672 }
673
674 // *** BUTTON EVENTS (101-106) ***
675 if ((eventId > 100) && (eventId < 200)) {
676 uint8_t wasOff = !ctx.isOn;
677 displayOnOff(ON, &ctx);
678
679 if (wasOff) {
680 vTaskResume(buttonTaskHandle);
681 continue; // Wake only, don't process
682 }
683
684 switch (ctx.state) {
685 case DISP_CLOCK: handleClockBtns(eventId, &ctx, buf); break;
686 case DISP_SET_RTC: handleSetRtcBtns(eventId, &ctx); break;
687 case DISP_SET_SILENT: handleSetSilentBtns(eventId, &ctx, buf); break;
688 case DISP_SET_CORRECTION: handleSetCorrBtns(eventId, &ctx, buf); break;
689 case DISP_ERROR: break;
690 case DISP_SYNC: break;
691 }
692
693 if (ctx.state != DISP_SYNC) {
694 vTaskResume(buttonTaskHandle);
695 }
696 }
697
698 // *** SYNC EVENTS (201-206) ***
699 if ((eventId > 200) && (eventId < 300)) {
700 ctx.state = DISP_SYNC;
701 displayOnOff(ON, &ctx);
702
703 if (eventId == DISP_EV_SYN_START) {
704 ssd1306_ClearScreen();
705 displayTitle(ctx.state, buf);
706 }
707
708 displayMessage(eventId - DISP_EV_SYN_START, buf);
709
710 if (eventId == DISP_EV_SYN_END) {
711 ctx.state = DISP_CLOCK;
712 vTaskDelay(pdMS_TO_TICKS(1000));
713 ssd1306_ClearScreen();
714 displayOnOff(ON, &ctx);
715 ctx.lastClockUpdate = xTaskGetTickCount() - pdMS_TO_TICKS(DISPLAY_CLOCK_INTERVAL) - 10;
716 vTaskResume(buttonTaskHandle);
717 }
718 }
719
720 // *** ERROR EVENTS (301-309) ***
721 if ((eventId > 300) && (eventId < 400)) {
722 ctx.state = DISP_ERROR;
723 displayOnOff(ON, &ctx);
724 displayTitle(ctx.state, buf);
725 displayMessage(eventId - DISP_EV_ERR_START, buf);
726 vTaskResume(buttonTaskHandle);
727 }
728
729 } else { // Event wait timed out
730
731 // Time clock display
732 if (ctx.state == DISP_CLOCK) {
733 if (timeLapsed(xTaskGetTickCount(), ctx.lastClockUpdate) > pdMS_TO_TICKS(DISPLAY_CLOCK_INTERVAL)) {
734 displayTitle(DISP_CLOCK, buf);
735 displayUpdateTimeVar(ctx.showTime);
736 displayShowClock(ctx.showTime);
737 ctx.lastClockUpdate = xTaskGetTickCount();
738 }
739 }
740
741 // Display auto shut off
742 if (ctx.state != DISP_SYNC) {
743 if (timeLapsed(xTaskGetTickCount(), ctx.lastOnTime) > pdMS_TO_TICKS(DISPLAY_OFF_TIMEOUT)) {
744 displayOnOff(OFF, &ctx);
745 if (ctx.state != DISP_ERROR) {
746 ctx.state = DISP_CLOCK;
747 }
748 }
749 }
750 }
751
752 } // While loop end
753}
