118static void clockAdvMinute(const uint8_t slow) {
119 uint32_t coilExcite = COIL_EXCITE_TIME + (COIL_EXTRA_TIME * slow);
120 uint32_t coilRest = COIL_REST_TIME + (COIL_EXTRA_TIME * slow);
121 uint8_t tickType = getLastTick();
122
123 if (tickType == 0) {
124#ifndef CIFRA5_DEBUG
125 HAL_GPIO_WritePin(CLK_TICK_GPIO_Port, CLK_TICK_Pin, GPIO_PIN_RESET);
126 vTaskDelay(pdMS_TO_TICKS(coilExcite));
127 HAL_GPIO_WritePin(CLK_TICK_GPIO_Port, CLK_TICK_Pin, GPIO_PIN_SET);
128 vTaskDelay(pdMS_TO_TICKS(coilRest));
129#endif
130 tickType = 1;
131 } else {
132#ifndef CIFRA5_DEBUG
133 HAL_GPIO_WritePin(CLK_TOCK_GPIO_Port, CLK_TOCK_Pin, GPIO_PIN_RESET);
134 vTaskDelay(pdMS_TO_TICKS(coilExcite));
135 HAL_GPIO_WritePin(CLK_TOCK_GPIO_Port, CLK_TOCK_Pin, GPIO_PIN_SET);
136 vTaskDelay(pdMS_TO_TICKS(coilRest));
137#endif
138 tickType = 0;
139 }
140
141 // Update mechanical position and tick state in backup registers
142 incrementMechMinute();
143 setLastTick(tickType);
144}

173static void searchForZeroPosition(void) {
174 uint8_t i;
175 uint8_t prevSens;
176
177 // Search for 00 minutes using hour sensor
178 xTaskNotify(displayTaskHandle, (uint32_t) DISP_EV_SYN_SRC_HOUR, eSetValueWithOverwrite);
179 i = 0;
180 do {
181 if (i == 62) { // Error: sensor not found
182 xTaskNotify(displayTaskHandle, (uint32_t) DISP_EV_ERR_SNS_HOUR, eSetValueWithOverwrite);
183 vTaskSuspend(NULL);
184 }
185 prevSens = HAL_GPIO_ReadPin(SNS_HOUR_GPIO_Port, SNS_HOUR_Pin);
186 clockAdvMinute(0);
187 i++;
188 } while (!((prevSens == 0) && (HAL_GPIO_ReadPin(SNS_HOUR_GPIO_Port, SNS_HOUR_Pin) == 1)));
189
190 // Now at XX:00, search for 00 hours using day sensor
191 xTaskNotify(displayTaskHandle, (uint32_t) DISP_EV_SYN_SRC_DAY, eSetValueWithOverwrite);
192
193 // Start PWM and park servo
194 prepareServo();
195 i = 0;
196 do {
197 if (i == 25) { // Error: sensor not found
198 shutdownServo();
199 xTaskNotify(displayTaskHandle, (uint32_t) DISP_EV_ERR_SNS_DAY, eSetValueWithOverwrite);
200 vTaskSuspend(NULL);
201 }
202 prevSens = HAL_GPIO_ReadPin(SNS_DAY_GPIO_Port, SNS_DAY_Pin);
203 clockAdvHour();
204 i++;
205 } while (!((prevSens == 1) && (HAL_GPIO_ReadPin(SNS_DAY_GPIO_Port, SNS_DAY_Pin) == 0)));
206
207 // Reset mechanical position to 00:00
208 resetMechPosition();
209}

322void clockTask(void *parameters) {
323 uint8_t rtcInitOk = (uint8_t)(uintptr_t) parameters; // When = 0 the clock has NOT been initialized
324 uint32_t message;
325 uint8_t prevSens;
326 uint8_t syncCount = 0;
327 uint8_t inSilentMode = 0;
328
329 while (1) {
330
331 // ===== PHASE 1: PRE-SYNC =====
332
333 // First boot: RTC not initialized, let user set time first
334 if (!rtcInitOk) {
335 rtcInitOk = 1;
336 xTaskNotify(displayTaskHandle, (uint32_t) DISP_EV_FORCE_SETUP, eSetValueWithOverwrite);
337 xTaskNotifyWait(0xffffffff, 0xffffffff, &message, portMAX_DELAY);
338 // User finished setting time, proceed to sync
339 }
340
341 // Wait out silent period (power outage recovery)
342 while (isInSilentPeriod()) {
343 vTaskDelay(pdMS_TO_TICKS(60000));
344 }
345
346 // ===== PHASE 2: SYNC =====
347
348 vTaskSuspend(buttonTaskHandle); // Disable buttons during sync (no-op if already suspended)
349 vTaskDelay(pdMS_TO_TICKS(200)); // Wait to complete any display action
350
351 if (syncCount == 3) { // Too many resynchronizations
352 xTaskNotify(displayTaskHandle, (uint32_t) DISP_EV_ERR_MANY_SYNC, eSetValueWithOverwrite);
353 }
354
355 xTaskNotify(displayTaskHandle, (uint32_t) DISP_EV_SYN_START, eSetValueWithOverwrite);
356 vTaskDelay(pdMS_TO_TICKS(1000)); // Wait to show the message on display
357
358 // Check if mechanical position is at 0:0 (first time sync needs sensor search)
359 if (getMechHours() == 0 && getMechMinutes() == 0) {
360 // FIRST TIME SYNC: Use sensor-based search to find 00:00
361 searchForZeroPosition();
362 } else {
363 // FAST SYNC: Position known from backup registers
364 advanceToHourBoundary();
365 }
366
367 HAL_RTC_GetTime(hrtcHandle, &RTC_Time, RTC_FORMAT_BIN); // Get updated time
368 HAL_RTC_GetDate(hrtcHandle, &RTC_Date, RTC_FORMAT_BIN);
369
370 syncHours(RTC_Time.Hours);
371 syncMinutes(RTC_Time.Minutes);
372
373 xTaskNotify(displayTaskHandle, (uint32_t) DISP_EV_SYN_END, eSetValueWithOverwrite);
374
375 // ===== PHASE 3: NORMAL OPERATION =====
376
377 inSilentMode = 0;
378
379 while (1) {
380 // Check for new time set by user
381 if (xTaskNotifyWait(0, 0xffffffff, &message, pdMS_TO_TICKS(CLOCK_UPDATE_INTERVAL)) == pdTRUE) {
382 if (message == CLOCK_EV_NEW_TIME) {
383 break; // User set new time -> resync
384 }
385 continue; // Ignore unexpected notifications
386 }
387
388 // Silent period entry
389 if (isInSilentPeriod()) {
390 inSilentMode = 1;
391 continue;
392 }
393
394 // Silent period exit -> resync
395 if (inSilentMode) {
396 inSilentMode = 0;
397 break;
398 }
399
400 // Check if minute advance needed
401 if (RTC_Time.Minutes == getMechMinutes()) {
402 continue;
403 }
404
405 prevSens = HAL_GPIO_ReadPin(SNS_HOUR_GPIO_Port, SNS_HOUR_Pin);
406 clockAdvMinute(1);
407
408 // Unexpected hour transition: mechanical drift detected, force full resync
409 if ((prevSens == 0)
410 && (HAL_GPIO_ReadPin(SNS_HOUR_GPIO_Port, SNS_HOUR_Pin) == 1)
411 && (getMechMinutes() != 0)) {
412 resetMechPosition();
413 syncCount++;
414 break;
415 }
416 }
417
418 } // Outer while loop end
419}
