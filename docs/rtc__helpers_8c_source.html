284void flashWriteSettings(void) {
285 uint8_t silentStart = getSilentStartHour();
286 uint8_t silentEnd = getSilentEndHour();
287 uint8_t calibPlus;
288 uint16_t calibValue;
289 getCalibration(&calibPlus, &calibValue);
290
291 uint32_t calibRaw = (calibValue & 0x1FF) | (calibPlus ? 0x200 : 0);
292 uint32_t word0 = (uint32_t)silentStart
293 | ((uint32_t)silentEnd << 8)
294 | (calibRaw << 16);
295 uint64_t data = (uint64_t)word0 | ((uint64_t)FLASH_SETTINGS_MAGIC << 32);
296
297 FLASH_EraseInitTypeDef eraseInit;
298 uint32_t pageError;
299
300 taskENTER_CRITICAL();
301
302 HAL_FLASH_Unlock();
303
304 eraseInit.TypeErase = FLASH_TYPEERASE_PAGES;
305 eraseInit.Page = FLASH_SETTINGS_PAGE;
306 eraseInit.NbPages = 1;
307 HAL_FLASHEx_Erase(&eraseInit, &pageError);
308
309 HAL_FLASH_Program(FLASH_TYPEPROGRAM_DOUBLEWORD, FLASH_SETTINGS_ADDR, data);
310
311 HAL_FLASH_Lock();
312
313 taskEXIT_CRITICAL();
314}

328void flashRestoreSettings(void) {
329 uint32_t *addr = (uint32_t *)FLASH_SETTINGS_ADDR;
330 uint32_t word0 = addr[0];
331 uint32_t word1 = addr[1];
332
333 if (word1 != FLASH_SETTINGS_MAGIC) {
334 // Fresh device: write defaults to backup registers
335 setSilentHours(SILENT_DEFAULT_START, SILENT_DEFAULT_END);
336 setCalibration(0, 0);
337 return;
338 }
339
340 uint8_t silentStart = (uint8_t)(word0 & 0xFF);
341 uint8_t silentEnd = (uint8_t)((word0 >> 8) & 0xFF);
342 uint16_t calibRaw = (uint16_t)((word0 >> 16) & 0x3FF);
343
344 if (silentStart <= 23 && silentEnd <= 23) {
345 setSilentHours(silentStart, silentEnd);
346 }
347
348 uint8_t calibPlus = (calibRaw & 0x200) ? 1 : 0;
349 uint16_t calibVal = calibRaw & 0x1FF;
350 setCalibration(calibPlus, calibVal);
351}

376uint8_t isInSilentPeriod(void) {
377 HAL_RTC_GetTime(hrtcHandle, &RTC_Time, RTC_FORMAT_BIN);
378 HAL_RTC_GetDate(hrtcHandle, &RTC_Date, RTC_FORMAT_BIN);
379 uint8_t start = getSilentStartHour();
380 uint8_t end = getSilentEndHour();
381 uint8_t pastStart = (RTC_Time.Hours > start)
382 || (RTC_Time.Hours == start && RTC_Time.Minutes > 0);
383 if (start > end) {
384 // Wraps around midnight (e.g., 22:01 to 09:00)
385 return (pastStart || RTC_Time.Hours < end);
386 } else {
387 // Same day (e.g., 02:01 to 05:00)
388 return (pastStart && RTC_Time.Hours < end);
389 }
390}
